What is $.messageBox?

An WinForm MessageBox class equivalent in JavaScript based on jQueryUI. Download it here. Enjoy!
- Duy A. Nguyen

Required set up

CSS

                <style type="text/css">
                    /* import jquery ui style */
                    @import url("http://ajax.googleapis.com/ajax/libs/jqueryui/1.8.14/themes/ui-darkness/jquery-ui.css");

                    /* import messageBox style */
                    @import url("messageBox.css");
                </style>

Javascript

                <!-- jquery & jqueryUI include-->
                <script type="text/javascript" src="https://ajax.googleapis.com/ajax/libs/jquery/1.6.2/jquery.min.js"></script> 
                <script type="text/javascript" src="https://ajax.googleapis.com/ajax/libs/jqueryui/1.8.14/jquery-ui.min.js"></script>    
                <!-- messageBox include -->
                <script type="text/javascript" src="messageBox.js"></script>

Examples

Default

                $.messageBox.show("Your text goes here");

Alert Dialog

                $.messageBox.show({
                    caption: "Your title goes here",
                    text: "Your text goes here",
                    ok: function(){
                        alert('You click OK');
                    }
                });

Confirm Dialog

                $.messageBox.show({
                        caption: "Confirm",
                        text: "Are you sure you want to delete this record?",
                        buttons: $.messageBox.Buttons.YesNo,
                        icon: $.messageBox.Icon.Exclamation,
                        yes: function(){
                            alert('You click YES');
                        },
                        no: function(){
                            alert('You click NO');
                        }
                    });

Abort-Retry-Ignore Dialog

                $.messageBox.show({
                        caption: "Error",
                        text: "Could not delete file ABC.pdf since it is currently inused. What do you want to do next?",
                        buttons: $.messageBox.Buttons.AbortRetryIgnore,
                        icon: $.messageBox.Icon.Error,
                        abort: function(){
                            alert('You click Abort');
                        },
                        retry: function(){
                            alert('You click Retry');
                        },
                        ignore: function(){
                            alert('You click Ignore');
                        }
                  });
